COMMENT(#,Install the latest miniconda environment and add it to the path)
RUN curl -o miniconda.sh http://repo.continuum.io/miniconda/Miniconda-latest-Linux-x86.sh && \
    chmod +x miniconda.sh && ./miniconda.sh -b && rm miniconda.sh && \
    echo "export PATH=${HOME}/miniconda3/bin:${HOME}/miniconda/bin:$PATH" >> ${HOME}/.bash_profile
env PATH ${HOME}/miniconda3/bin:${HOME}/miniconda/bin:$PATH
